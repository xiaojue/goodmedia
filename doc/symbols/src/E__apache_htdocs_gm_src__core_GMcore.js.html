1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview Goodmedia前端js库core文件，负责创建GM对象和构建命名空间 基于jquery1.6.2
  5  */
  6 (function(W,doc,$){
  7 	/**
  8 	 * @namespace
  9 	 * @name GM
 10 	 * @description 全局的GM对象
 11 	 */
 12 	
 13 	/**
 14 	 * @namespace
 15 	 * @name jQuery
 16 	 * @description 这里只对私自扩展的jQuery进行doc描述
 17 	 */
 18 	
 19 	if(!W.GM) var GM={};
 20 	/**
 21 	 * @namespace
 22 	 * @description 与业务紧密相关的-挂件
 23 	 */
 24 	GM.widget={};
 25 	/**
 26 	 * @namespace
 27 	 * @description 与业务无关的比如overlay组件,最后扩展到jquery上，使用jquery的扩展机制进行最后的封装
 28 	 */
 29 	GM.tools={};
 30 	/**
 31 	 * @namespace
 32 	 * @description 独立项目或者应用
 33 	 */
 34 	GM.apps={};
 35 	/**
 36 	 * @constant
 37 	 * @description 根据href里德debug关键字确定是否为debug模式
 38 	 */
 39 	GM.debug=W.GMDEBUG.aver;
 40 	/**
 41 	 * @constant
 42 	 * @description 获取当前js的根目录
 43 	 */
 44 	GM.host=function(){
 45 		var scripts=doc.getElementsByTagName('script'),i,base;
 46 		for(i=0;i<scripts.length;i++){
 47 			var src=scripts[i].src,
 48 					namereg=/(GM-min|GM).js/;
 49 			if(namereg.test(src)){
 50 				base=src.slice(0,src.lastIndexOf('/')+1);
 51 			}
 52 		}
 53 		return base;
 54 	}();
 55 	/**
 56 	 * @static
 57 	 * @description 转换到本地非压缩路径
 58 	 * @function
 59 	 * @param {string} uri
 60 	 * @returns {string} 处理过的路径
 61 	 */
 62 	GM.locality=function(uri){
 63 		return uri.replace(/bulid/g,'src').replace(/\-min/g,'');
 64 	}
 65 	/**
 66 	 * @constant
 67 	 * @description widget的根目录
 68 	 */
 69 	GM.widget.host=GM.host + 'widget/';
 70 	/**
 71 	 * @constant
 72 	 * @description apps的根目录
 73 	 */
 74 	GM.apps.host=GM.host + 'apps/';
 75 	
 76 	//debug模式下处理路径
 77 	if(GM.debug){
 78 		GM.host=W.GMDEBUG.host+'bulid/';
 79 		GM.widget.host=GM.locality(GM.host) + 'widget/';
 80 		GM.apps.host=GM.locality(GM.host) + 'apps/';
 81 	}
 82 	/**
 83 	 * @static
 84 	 * @description 额外加载项目文件 - 项目文件目前依赖关系依靠ant维护
 85 	 * @function
 86 	 * @param {string} appname
 87 	 * @param {function} callback
 88 	 */
 89 	
 90 	GM.apps.map={};
 91 
 92 	GM.apps.require=function(appname,callback){
 93 		if(GM.apps.map.hasOwnProperty(appname)){
 94 			callback(GM.apps[appname]['exports']);
 95 			return;
 96 		}
 97 		var appuri = GM.host + 'apps/'+appname+'/'+appname+'-min.js';
 98 		if(GM.debug) appuri=GM.locality(appuri);
 99 		$(function(){
100 			$.getScript(appuri,function(){
101 				if(callback){
102 					GM.apps.map[appname]={
103 						uri:appuri
104 					};
105 					callback(GM.apps[appname]['exports']);
106 				}
107 			});
108 		});
109 	}
110 	/**
111 	 * @static
112 	 * @description 把用过的widget储存，再次use则不再调用
113 	 * @private
114 	 */
115 	GM.widget.usemap={};
116 	/**
117 	 * @static
118 	 * @description 加载widget的方法
119 	 * @function
120 	 * @param {string} widget
121 	 * @param {function} callback
122 	 */
123 	GM.widget.use=function(widget,callback){
124 		if(GM.widget.usemap.hasOwnProperty(widget)){
125 			callback(GM.widget);
126 			return;
127 		}
128 		
129 		var widgeturi = GM.host + 'widget/'+widget+'/'+widget+'-min.js';
130 		if(GM.debug) widgeturi=GM.locality(widgeturi);
131 		$(function(){
132 			$.getScript(widgeturi,function(){
133 					GM.widget.usemap[widget]={
134 						uri:widgeturi
135 					}
136 					callback(GM.widget);
137 			});
138 		});
139 	}
140 	
141 	//debug 模式下开启debug.js并且初始化debug面板
142 	if(GM.debug){
143 		GM.widget.use('debug',function(widget){
144 			widget.debug.init();
145 		});
146 	}
147 	
148 	GM.apps.require('idongmi',function(exports){
149 		exports.init();
150 	});
151 
152 	$(function(){
153 	if($('#J_MuenbarV2').length>0){
154 		GM.apps.require('cmsmuenbarv2',function(exports){
155 			exports.init();
156 		});
157 	}
158 	})	
159 	
160 	W.GM=GM;
161 })(window,document,jQuery);
